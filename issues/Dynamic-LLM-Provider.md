Here's a design for a decoupled, config-driven LLM system:                                                                                                                                                                   
                                                                                                                                                                                                                               
  Proposed Architecture                                                                                                                                                                                                        
                                                                                                                                                                                                                               
  .env (or environment)                                                                                                                                                                                                        
      ↓                                                                                                                                                                                                                        
  app/config.py (Global LLM settings)                                                                                                                                                                                          
      ↓                                                                                                                                                                                                                        
  app/common/providers/llm.py (Provider factory reads config)                                                                                                                                                                  
      ↓                                                                                                                                                                                                                        
  platforms/*/config.py (Optional override)                                                                                                                                                                                    
      ↓                                                                                                                                                                                                                        
  agents/*/llm.py (Uses platform or global config)                                                                                                                                                                             
                                                                                                                                                                                                                               
  Configuration Design                                                                                                                                                                                                         
                                                                                                                                                                                                                               
  1. Global Config (.env)                                                                                                                                                                                                      
                                                                                                                                                                                                                               
  # Switch provider with one line                                                                                                                                                                                              
  LLM_PROVIDER=ollama              # ollama | openai | aws_bedrock                                                                                                                                                             
                                                                                                                                                                                                                               
  # Provider-specific settings                                                                                                                                                                                                 
  LLM_DEFAULT_MODEL=llama3         # Model name for chosen provider                                                                                                                                                            
  LLM_TEMPERATURE=0.7                                                                                                                                                                                                          
  LLM_MAX_TOKENS=1000                                                                                                                                                                                                          
                                                                                                                                                                                                                               
  # Provider credentials (only needed for respective provider)                                                                                                                                                                 
  OLLAMA_BASE_URL=http://localhost:11434                                                                                                                                                                                       
  OPENAI_API_KEY=sk-...                                                                                                                                                                                                        
  AWS_REGION=us-east-1                                                                                                                                                                                                         
  AWS_BEDROCK_MODEL=anthropic.claude-v2                                                                                                                                                                                        
                                                                                                                                                                                                                               
  2. Switching Providers                                                                                                                                                                                                       
  ┌──────────┬──────────────────────────────────────┐                                                                                                                                                                          
  │ Provider │           Required Config            │                                                                                                                                                                          
  ├──────────┼──────────────────────────────────────┤                                                                                                                                                                          
  │ Ollama   │ LLM_PROVIDER=ollama, OLLAMA_BASE_URL │                                                                                                                                                                          
  ├──────────┼──────────────────────────────────────┤                                                                                                                                                                          
  │ OpenAI   │ LLM_PROVIDER=openai, OPENAI_API_KEY  │                                                                                                                                                                          
  ├──────────┼──────────────────────────────────────┤                                                                                                                                                                          
  │ Bedrock  │ LLM_PROVIDER=aws_bedrock, AWS_REGION │                                                                                                                                                                          
  └──────────┴──────────────────────────────────────┘                                                                                                                                                                          
  3. Platform Override (Optional)                                                                                                                                                                                              
                                                                                                                                                                                                                               
  # platforms/hello_world/config.py                                                                                                                                                                                            
  class HelloWorldConfig(BaseSettings):                                                                                                                                                                                        
      # Override global LLM for this platform only                                                                                                                                                                             
      LLM_PROVIDER: str | None = None  # None = use global                                                                                                                                                                     
      LLM_MODEL: str | None = None     # None = use global                                                                                                                                                                     
                                                                                                                                                                                                                               
  4. Code Flow                                                                                                                                                                                                                 
                                                                                                                                                                                                                               
  # Agent gets LLM without knowing which provider                                                                                                                                                                              
  llm = get_llm_provider()  # Reads from config automatically                                                                                                                                                                  
                                                                                                                                                                                                                               
  # Or platform-specific override                                                                                                                                                                                              
  llm = get_llm_provider(                                                                                                                                                                                                      
      provider_type=platform_config.LLM_PROVIDER,  # Optional override                                                                                                                                                         
      model=platform_config.LLM_MODEL                                                                                                                                                                                          
  )                                                                                                                                                                                                                            
                                                                                                                                                                                                                               
  ---                                                                                                                                                                                                                          
  Files to Modify                                                                                                                                                                                                              
  ┌─────────────────────────────────────────────┬───────────────────────────────────────────┐                                                                                                                                  
  │                    File                     │                  Change                   │                                                                                                                                  
  ├─────────────────────────────────────────────┼───────────────────────────────────────────┤                                                                                                                                  
  │ app/config.py                               │ Add LLM_PROVIDER, LLM_DEFAULT_MODEL, etc. │                                                                                                                                  
  ├─────────────────────────────────────────────┼───────────────────────────────────────────┤                                                                                                                                  
  │ common/providers/llm.py                     │ Update factory to read from config        │                                                                                                                                  
  ├─────────────────────────────────────────────┼───────────────────────────────────────────┤                                                                                                                                  
  │ platforms/hello_world/config.py             │ Add optional LLM override                 │                                                                                                                                  
  ├─────────────────────────────────────────────┼───────────────────────────────────────────┤                                                                                                                                  
  │ platforms/hello_world/agents/greeter/llm.py │ Use config-driven provider                │                                                                                                                                  
  └─────────────────────────────────────────────┴───────────────────────────────────────────┘                                                                                                                                  
  ---                                                                                                                                                                                                                          
  Example Usage After Implementation                                                                                                                                                                                           
                                                                                                                                                                                                                               
  Using Ollama (local):                                                                                                                                                                                                        
  LLM_PROVIDER=ollama                                                                                                                                                                                                          
  LLM_DEFAULT_MODEL=llama3                                                                                                                                                                                                     
                                                                                                                                                                                                                               
  Switch to OpenAI:                                                                                                                                                                                                            
  LLM_PROVIDER=openai                                                                                                                                                                                                          
  LLM_DEFAULT_MODEL=gpt-4                                                                                                                                                                                                      
  OPENAI_API_KEY=sk-xxx                                                                                                                                                                                                        
                                                                                                                                                                                                                               
  Switch to AWS Bedrock:                                                                                                                                                                                                       
  LLM_PROVIDER=aws_bedrock                                                                                                                                                                                                     
  LLM_DEFAULT_MODEL=anthropic.claude-v2                                                                                                                                                                                        
  AWS_REGION=us-east-1
