  Model Schemas                                                                                                                                                                                                                
                                                                                                                                                                                                                               
  Newsletter                                                                                                                                                                                                                   
  - id, user_id, title, content, html_content, plain_text                                                                                                                                                                      
  - status (draft/generating/ready/sent/failed)                                                                                                                                                                                
  - topics_covered, tone_used, word_count, read_time                                                                                                                                                                           
  - research_data, writing_data, mindmap_markdown                                                                                                                                                                              
  - created_at, sent_at                                                                                                                                                                                                        
                                                                                                                                                                                                                               
  Subscriber                                                                                                                                                                                                                   
  - id, email, name, status (subscribed/unsubscribed/bounced)                                                                                                                                                                  
  - preferences (topics, tone, frequency)                                                                                                                                                                                      
  - engagement_metrics, tags                                                                                                                                                                                                   
  - subscribed_at, unsubscribed_at                                                                                                                                                                                             
                                                                                                                                                                                                                               
  Campaign                                                                                                                                                                                                                     
  - id, name, subject, template_id                                                                                                                                                                                             
  - status (draft/scheduled/sending/sent/failed)                                                                                                                                                                               
  - recipient_count, open_count, click_count                                                                                                                                                                                   
  - scheduled_at, sent_at                                                                                                                                                                                                      
                                                                                                                                                                                                                               
  Template                                                                                                                                                                                                                     
  - id, name, content, variables, category                                                                                                                                                                                     
  - created_at, updated_at
